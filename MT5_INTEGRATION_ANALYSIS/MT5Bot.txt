input double InitialEquityInput = 0.0; // If >0, overrides account equity at startup



int OnInit()

{

   // Capture initial equity exactly once

   if(InitialEquityInput > 0.0)

      initialEquity = InitialEquityInput;

   else

      initialEquity = AccountInfoDouble(ACCOUNT_EQUITY);



   maxEquity        = initialEquity;

   drawdownLimit    = initialEquity * 0.92;   // 8% daily drawdown limit

   maxDrawdownLimit = initialEquity * 0.88;   // 12% overall drawdown limit



   Print("EA applied. Initial equity: ", DoubleToString(initialEquity,2));

   Print("Daily drawdown limit set to: ", DoubleToString(drawdownLimit,2));

   Print("Max drawdown limit set to: ", DoubleToString(maxDrawdownLimit,2));



   if(!drawdownBreached)

   {

      SendInitialEquityEmail();

      SendDrawdownLimitNotification();

   }



   // Schedule OnTimer every UPDATE_INTERVAL seconds

   EventSetTimer(UPDATE_INTERVAL);



   // Send the very first account update

   SendAccountUpdate();



   return(INIT_SUCCEEDED);

}

//| Timer function: reset daily values and email flags               |



//+------------------------------------------------------------------+



void OnTimer()



{



   datetime now = TimeCurrent();



   MqlDateTime currentTimeStruct;



   TimeToStruct(now, currentTimeStruct);







   // Reset daily drawdown limit at market close (example: 21:05) if not breached



   if(currentTimeStruct.hour == 21 && currentTimeStruct.min == 5 && !dailyDrawdownEmailSent && !drawdownBreached)



   {



      SetDailyDrawdownLimit();



      dailyDrawdownEmailSent = true;



   }







   // Reset email flag after midnight (example: 00:10)



   if(currentTimeStruct.hour == 0 && currentTimeStruct.min == 10)



      dailyDrawdownEmailSent = false;



}

//| Get trade history and format as JSON array (Loop-Safe Version)   |
//+------------------------------------------------------------------+
string GetRecentTradesJSON()
{
   // Select last 30 days of history
   if(!HistorySelect(TimeCurrent() - (30 * 86400), TimeCurrent() + 3600)) return "[]";
   
   int total = HistoryDealsTotal();
   string json = "[";
   int count = 0;
   
   // Loop 1: Find all 'OUT' deals first
   for(int i = 0; i < total; i++)
   {
      ulong ticket = HistoryDealGetTicket(i);
      if(ticket <= 0) continue;
      
      long entry = HistoryDealGetInteger(ticket, DEAL_ENTRY);
      
      if(entry == DEAL_ENTRY_OUT || entry == DEAL_ENTRY_INOUT || entry == DEAL_ENTRY_OUT_BY)
      {
         long pos_id = HistoryDealGetInteger(ticket, DEAL_POSITION_ID);
         if(pos_id <= 0) continue;

         double close_price = HistoryDealGetDouble(ticket, DEAL_PRICE);
         double profit = HistoryDealGetDouble(ticket, DEAL_PROFIT);
         double volume = HistoryDealGetDouble(ticket, DEAL_VOLUME);
         datetime close_time = (datetime)HistoryDealGetInteger(ticket, DEAL_TIME);
         string symbol = HistoryDealGetString(ticket, DEAL_SYMBOL);
         long magic = HistoryDealGetInteger(ticket, DEAL_MAGIC);
         string comment = HistoryDealGetString(ticket, DEAL_COMMENT);

         // Search for the corresponding 'IN' deal in the ALREADY SELECTED history
         double open_price = close_price;
         datetime open_time = close_time;
         string type_str = "BUY";
         
         for(int j = 0; j < total; j++)
         {
            ulong t_in = HistoryDealGetTicket(j);
            if(HistoryDealGetInteger(t_in, DEAL_POSITION_ID) == pos_id && 
               HistoryDealGetInteger(t_in, DEAL_ENTRY) == DEAL_ENTRY_IN)
            {
               open_price = HistoryDealGetDouble(t_in, DEAL_PRICE);
               open_time = (datetime)HistoryDealGetInteger(t_in, DEAL_TIME);
               long original_type = HistoryDealGetInteger(t_in, DEAL_TYPE);
               type_str = (original_type == DEAL_TYPE_BUY) ? "BUY" : "SELL";
               break;
            }
         }

         string s_open_time = TimeToString(open_time, TIME_DATE|TIME_SECONDS);
         string s_close_time = TimeToString(close_time, TIME_DATE|TIME_SECONDS);
         StringReplace(s_open_time, ".", "-");
         StringReplace(s_close_time, ".", "-");

         if(count > 0) json += ",";
         json += "{";
         json += "\"ticket\":\"" + IntegerToString(ticket) + "\",";
         json += "\"symbol\":\"" + symbol + "\",";
         json += "\"type\":\"" + type_str + "\",";
         json += "\"volume\":" + DoubleToString(volume, 2) + ",";
         json += "\"open_price\":" + DoubleToString(open_price, _Digits) + ",";
         json += "\"close_price\":" + DoubleToString(close_price, _Digits) + ",";
         json += "\"profit\":" + DoubleToString(profit, 2) + ",";
         json += "\"open_time\":\"" + s_open_time + "\",";
         json += "\"close_time\":\"" + s_close_time + "\",";
         json += "\"magic\":" + IntegerToString(magic) + ",";
         json += "\"comment\":\"" + comment + "\"";
         json += "}";
         count++;
      }
      
      // Safety cap: don't send more than 50 trades in one go
      if(count >= 50) break;
   }
   
   json += "]";
   return json;
}

bool isBreached = false;

//+------------------------------------------------------------------+
//| Hard Lockdown: Guardian against further trading                  |
//+------------------------------------------------------------------+
void HandleBreachEnforcement(string reason)
{
   isBreached = true;
   Print("CRITICAL: ACCOUNT BREACHED. Reason: ", reason);
   ChartSetSymbolPeriod(0, _Symbol, _Period); // Refresh chart
   Comment("ACCOUNT BREACHED: " + reason + "\nTrading is disabled.");
   
   // 1. Close all open positions immediately
   for(int i=PositionsTotal()-1; i>=0; i--)
   {
      ulong ticket = PositionGetTicket(i);
      if(ticket > 0)
      {
         CTrade trade_exec;
         trade_exec.PositionClose(ticket);
      }
   }
   
   // 2. Delete all pending orders
   for(int i=OrdersTotal()-1; i>=0; i--)
   {
      ulong ticket = OrderGetTicket(i);
      if(ticket > 0)
      {
         CTrade trade_exec;
         trade_exec.OrderDelete(ticket);
      }
   }
}

//| Enhanced SendAccountUpdate with Enforcement Feedback             |
//+------------------------------------------------------------------+
void SendAccountUpdate()
{
   Print("SYNC: Starting account update...");
   string accId = IntegerToString(AccountInfoInteger(ACCOUNT_LOGIN));
   string trades_json = GetRecentTradesJSON();
   
   Print("SYNC: Prepared JSON with trade data length: ", StringLen(trades_json));
   
   string json = "{";
   json += "\"mt5_id\":\"" + accId + "\",";
   json += "\"client_name\":\"" + clientName + "\",";
   json += "\"initial_equity\":" + DoubleToString(initialEquity, 2) + ",";
   json += "\"daily_drawdown_limit\":" + DoubleToString(drawdownLimit, 2) + ",";
   json += "\"overall_drawdown_limit\":" + DoubleToString(maxDrawdownLimit, 2) + ",";
   json += "\"running_balance\":" + DoubleToString(AccountInfoDouble(ACCOUNT_BALANCE), 2) + ",";
   json += "\"running_equity\":" + DoubleToString(AccountInfoDouble(ACCOUNT_EQUITY), 2) + ",";
   json += "\"trades\":" + trades_json;
   json += "}";

   string headers = "Content-Type: application/json\r\n";
   char postData[];
   char result[];
   string resultHeaders;
   StringToCharArray(json, postData, 0, StringLen(json));

   int res = WebRequest("POST", PHP_ENDPOINT, headers, 5000, postData, result, resultHeaders);
   
   if(res == 200)
   {
      string response = CharArrayToString(result);
      Print("SYNC SUCCESS. Server Response: ", response);
      
      if(StringFind(response, "\"status\":\"breached\"") >= 0)
      {
         if(!isBreached) { // Only call this once if it transitions
            HandleBreachEnforcement("Rule violation detected by Strategy Warden.");
         }
      }
   }
   else if(res == -1) 
   {
      Print("SYNC ERROR: WebRequest failed. Error code: ", GetLastError());
   }
   else 
   {
      Print("SYNC HTTP ERROR: Server returned status code ", res);
   }
}



//+------------------------------------------------------------------+



//| Expert tick function                                             |



//+------------------------------------------------------------------+



void OnTick()
{
   // Moved SendAccountUpdate timer to the top of OnTick
   static datetime lastUpdateTime = 0;
   if(TimeTradeServer() - lastUpdateTime >= UPDATE_INTERVAL)
   {
      SendAccountUpdate();
      lastUpdateTime = TimeTradeServer();
   }

   if(isBreached)
   {
      // Extra protection: keep closing positions if they are opened manually
      if(PositionsTotal() > 0 || OrdersTotal() > 0) HandleBreachEnforcement("Persistent lockdown active.");
      return;
   }

   double currentEquity = AccountInfoDouble(ACCOUNT_EQUITY);



   // Update maximum equity if current equity is higher

   if(currentEquity > maxEquity)

   {

      maxEquity = currentEquity;



      // Only print the first new daily high once per day

      static int lastHighPrintDate = 0;

      MqlDateTime dt; TimeToStruct(TimeCurrent(), dt);

      int today = dt.year * 10000 + dt.mon * 100 + dt.day;



      if(today != lastHighPrintDate)

      {

         Print("New daily high equity: ", DoubleToString(maxEquity,2));

         lastHighPrintDate = today;

      }

   }





   // Check drawdown conditions



   if(!tradingStopped && currentEquity <= drawdownLimit)



   {



      Print("Daily drawdown limit breached! Current Equity: ", DoubleToString(currentEquity,2));



      Alert("Daily drawdown limit breached! Current Equity: " + DoubleToString(currentEquity,2));



      NotifyDrawdownBreach();  // Sends emails/notifications and updates trading_accounts



      CloseAllPositions();



      tradingStopped = true;



      drawdownBreached = true;



   }



   if(!tradingStopped && currentEquity <= maxDrawdownLimit)



   {



      Print("Max drawdown limit breached! Current Equity: ", DoubleToString(currentEquity,2));



      Alert("Max drawdown limit breached! Current Equity: " + DoubleToString(currentEquity,2));



      NotifyMaxDrawdownBreach();  // Sends emails/notifications and updates trading_accounts



      CloseAllPositions();



      tradingStopped = true;



      drawdownBreached = true;



   }







   // Continuously close any new positions if trading is stopped
   if(tradingStopped)
      CloseAllNewPositions();
}

}

//| Close all open positions                                           |



//+------------------------------------------------------------------+



void CloseAllPositions()



{



   int totalPositions = PositionsTotal();



   Print("Closing all positions. Total positions: ", IntegerToString(totalPositions));



   Alert("Closing all positions. Total positions: " + IntegerToString(totalPositions));



   for(int i = totalPositions - 1; i >= 0; i--)



   {



      if(PositionSelect(PositionGetSymbol(i)))



      {



         ulong ticket = PositionGetInteger(POSITION_TICKET);



         bool result = trade.PositionClose(ticket);



         if(result)



         {



            Print("Position closed successfully. Ticket: ", ticket);



            Alert("Position closed successfully. Ticket: " + IntegerToString(ticket));



         }



         else



         {



            Print("Failed to close position. Error: ", GetLastError());



            Alert("Failed to close position. Error: " + IntegerToString(GetLastError()));



         }



      }



   }



}







//+------------------------------------------------------------------+



//| Continuously close new positions after trading is stopped        |



//+------------------------------------------------------------------+



void CloseAllNewPositions()



{



   int totalPositions = PositionsTotal();



   for(int i = totalPositions - 1; i >= 0; i--)



   {



      if(PositionSelect(PositionGetSymbol(i)))



      {



         ulong ticket = PositionGetInteger(POSITION_TICKET);



         bool result = trade.PositionClose(ticket);



         if(result)



         {



            Print("New position closed successfully. Ticket: ", ticket);



            Alert("New position closed successfully. Ticket: " + IntegerToString(ticket));



            string message = "Trading disabled for " + clientName +



                             ". New trade closed. Ticket ID: " + IntegerToString(ticket) + ".";



            Alert(message);



            SendNotification(message);



         }



         else



         {



            Print("Failed to close new position. Error: ", GetLastError());



            Alert("Failed to close new position. Error: " + IntegerToString(GetLastError()));



         }



      }



   }



}

   json += "\"status\":\"breached\",";

   json += "}";



   Print("Updating trading account status: ", json);







   char postData[];



   int data_len = StringLen(json);



   ArrayResize(postData, data_len);



   StringToCharArray(json, postData, 0, data_len, CP_UTF8);







   string headers = "Content-Type: application/json\r\n";



   char result[];



   int res = WebRequest("POST", TRADING_ACCOUNT_ENDPOINT, headers, 5000, postData, result, headers);



   if(res == -1)



   {



      int error = GetLastError();



      Print("Error updating trading account. Error code: ", error);



   }



   else



   {



      Print("Trading account updated successfully.");



      // Uncomment to print server response:



      // Print("Response: ", CharArrayToString(result));



   }



}
